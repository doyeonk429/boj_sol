# 두 큐 합 같게 만들기

### [ 문제 정보 ]
> **프로그래머스**: https://school.programmers.co.kr/learn/courses/30/lessons/118667
> 
> **난이도**: 3
>
> **유형**: #자료구조


### [ 문제 풀이 ]
```Swift
import Foundation

func solution(_ queue1:[Int], _ queue2:[Int]) -> Int {
    var total = queue1 + queue2
    var n = queue1.count
    
    var q1Idx = 0
    var q2Idx = n
    
    var q1Sum = queue1.reduce(0, +)
    var q2Sum = queue2.reduce(0, +)
    
    var count = 0
    let maxCount = n * 3
    
    while q1Sum != q2Sum {
        if count > maxCount {
            return -1
        }
        
        if q1Sum < q2Sum { // q2 -> q1
            if q2Idx < total.count {
                let num = total[q2Idx]
                q1Sum += num
                q2Sum -= num
                count += 1
                q2Idx += 1
            } else {
                return -1
            }
        } else { // q1 -> q2
            if q1Idx < total.count {
                let num = total[q1Idx]
                q1Sum -= num
                q2Sum += num
                count += 1
                q1Idx += 1
            } else {
                return -1
            }
        }
        
    }
    
    return count
}
```
